Linguagem em C
Enunciado: calcular a média de uma série de notas de alunos e determinar quantos alunos foram aprovados
(nota >= 60) e quantos foram reprovados.
#include <stdio.h>
int main() {
int numAlunos, i;
float nota, soma = 0, media;
int aprovados = 0, reprovados = 0;
// Entrada: número de alunos
printf("Digite o número de alunos: ");
scanf("%d", &numAlunos);
// Estrutura de repetição: entrada das notas e cálculo da soma
for (i = 0; i < numAlunos; i++) {
printf("Digite a nota do aluno %d: ", i + 1);
scanf("%f", &nota);
soma += nota;
// Condicional composta: contagem de aprovados e reprovados
if (nota >= 60) {
aprovados++;
} else {
reprovados++;
}
}
// Cálculo da média
if (numAlunos > 0) {
media = soma / numAlunos;
printf("A média da turma é: %.2f\n", media);
} else {
printf("Nenhum aluno informado.\n");
return 1;
}
// Saída: número de aprovados e reprovados
printf("Número de alunos aprovados: %d\n", aprovados);
printf("Número de alunos reprovados: %d\n", reprovados);
return 0;
}

Linguagem Java
Enunciado: calcular a média de uma série de notas de alunos e determinar quantos alunos foram aprovados
(nota >= 60) e quantos foram reprovados.
import java.util.Scanner;
public class MediaNotas {
public static void main(String[] args) {
Scanner scanner = new Scanner(System.in);
int numAlunos;
float nota, soma = 0, media;
int aprovados = 0, reprovados = 0;
// Entrada: número de alunos
System.out.print("Digite o número de alunos: ");
numAlunos = scanner.nextInt();
// Estrutura de repetição: entrada das notas e cálculo da soma
for (int i = 0; i < numAlunos; i++) {
System.out.print("Digite a nota do aluno " + (i + 1) + ": ");
nota = scanner.nextFloat();
soma += nota;
// Condicional composta: contagem de aprovados e reprovados
if (nota >= 60) {
aprovados++;
} else {
reprovados++;
}
}
// Cálculo da média
if (numAlunos > 0) {
media = soma / numAlunos;
System.out.printf("A média da turma é: %.2f\n", media);
} else {
System.out.println("Nenhum aluno informado.");
return; // Encerrar o programa se não houver alunos
}
// Saída: número de aprovados e reprovados
System.out.println("Número de alunos aprovados: " + aprovados);
System.out.println("Número de alunos reprovados: " + reprovados);
scanner.close();
}
}
