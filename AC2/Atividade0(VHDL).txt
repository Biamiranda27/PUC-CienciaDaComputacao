---------design.vhd---------

library ieee;
use ieee.std_logic_1164.all;

entity mult2x1 is //essa é a parte das entradas e saída
    port (
        e1  : in std_logic; //primeira entrada de dados
        e2  : in std_logic; //segunda entrada de dados
        sel : in std_logic; //sinal de seleção
        s   : out std_logic //única saída
    );
//como estava escrito anteriormente estava certo também, mas por preferência troquei a //escrita
end mult2x1;

architecture arch_mult2x1 of mult2x1 is
begin
    with sel select
        s <= e1 when '0', //não sei se o erro foi por copiar e colar do canvas, mas //ajustei as aspas para aspas simples
             e2 when others;
end arch_mult2x1;
