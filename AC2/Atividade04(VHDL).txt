library ieee;
use ieee.std_logic_1164.all;
entity meioSomador is
port(
a : in std_logic;
b : in std_logic;
soma : out std_logic;
carry : out std_logic
);
end meioSomador;
architecture arch_meioSomador of meioSomador is
begin
soma <= (a and not b) or (not a and b);
carry <= a and b;
end arch_meioSomador;
entity mux2x1 is
port (
e1 : in std_logic;
e2 : in std_logic;
sel : in std_logic;
s : out std_logic
);
end mux2x1;
architecture arch_mult2x1 of mux2x1 is
begin
with sel select
s <= e1 when '0',
e2 when others;
end arch_mult2x1;
entity somadorcompleto is
port (
a : in std_logic;
b : in std_logic;
ci : in std_logic;
s : out std_logic;
co : out std_logic;
op_and : out std_logic
);
end somadorcompleto;
architecture structural of somadorcompleto is
component meioSomador
port(
a : in std_logic;
b : in std_logic;
soma : out std_logic;
carry : out std_logic
);
end component;
signal s_soma1 : std_logic;
signal s_carry1 : std_logic;
signal s_soma2 : std_logic;
begin
MS1: meioSomador
port map (
a => a,
b => b,
soma => s_soma1,
carry => s_carry1
);
MS2: meioSomador
port map (
a => s_soma1,
b => ci,
soma => s,
carry => s_soma2
);
co <= s_carry1 or s_soma2;
op_and <= a and b; -- Saída op_and
end structural;
